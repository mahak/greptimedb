36/// The key stores table repartition metadata.
37/// Specifically, it records the relation between source and destination regions after a repartition operation is completed.
38/// This is distinct from the initial partitioning scheme of the table.
39/// For example, after repartition, a destination region may still hold files from a source region; this mapping should be updated once repartition is done.
40/// The GC scheduler uses this information to clean up those files (and removes this mapping if all files from the source region are cleaned).
41///
42/// The layout: `__table_repart/{table_id}`.
43#[derive(Debug, PartialEq)]
44pub struct TableRepartKey {
45    /// The unique identifier of the table whose re-partition information is stored in this key.
46    pub table_id: TableId,
47}
48
49impl TableRepartKey {
50    pub fn new(table_id: TableId) -> Self {
51        Self { table_id }
52    }
53
54    /// Returns the range prefix of the table repartition key.
55    pub fn range_prefix() -> Vec<u8> {
56        format!("{}/", TABLE_REPART_PREFIX).into_bytes()
57    }
58}
59
60impl MetadataKey<'_, TableRepartKey> for TableRepartKey {
61    fn to_bytes(&self) -> Vec<u8> {
62        self.to_string().into_bytes()
63    }
64
65    fn from_bytes(bytes: &[u8]) -> Result<TableRepartKey> {
66        let key = std::str::from_utf8(bytes).map_err(|e| {
67            InvalidMetadataSnafu {
68                err_msg: format!(
69                    "TableRepartKey '{}' is not a valid UTF8 string: {e}",
70                    String::from_utf8_lossy(bytes)
71                ),
72            }
73            .build()
74        })?;
75        let captures = TABLE_REPART_KEY_PATTERN
76            .captures(key)
77            .context(InvalidMetadataSnafu {
78                err_msg: format!("Invalid TableRepartKey '{key}'"),
79            })?;
80        // Safety: pass the regex check above
81        let table_id = captures[1].parse::<TableId>().unwrap();
82        Ok(TableRepartKey { table_id })
83    }
84}
85
86impl Display for TableRepartKey {
87    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
88        write!(f, "{}/{}", TABLE_REPART_PREFIX, self.table_id)
89    }
90}

92#[derive(Debug, PartialEq, Serialize, Deserialize, Clone, Default)]
93pub struct TableRepartValue {
94    /// A mapping from source region IDs to sets of destination region IDs after repartition.
95    ///
96    /// Each key in the map is a `RegionId` representing a source region that has been repartitioned.
97    /// The corresponding value is a `BTreeSet<RegionId>` containing the IDs of destination regions
98    /// that currently hold files originally from the source region. This mapping is updated after
99    /// repartition and is used by the GC scheduler to track and clean up files that have been moved.
100    pub src_to_dst: BTreeMap<RegionId, BTreeSet<RegionId>>,
101}
102
103impl TableRepartValue {
104    /// Creates a new TableRepartValue with an empty src_to_dst map.
105    pub fn new() -> Self {
106        Default::default()
107    }
108    /// Update mapping from src region to dst regions. Should be called once repartition is done.
109    ///
110    /// If `dst` is empty, this method does nothing.
111    pub fn update_mappings(&mut self, src: RegionId, dst: &[RegionId]) {
112        if dst.is_empty() {
113            return;
114        }
115        self.src_to_dst.entry(src).or_default().extend(dst);
116    }
117
118    /// Remove mappings from src region to dst regions. Should be called once files from src region are cleaned up in dst regions.
119    pub fn remove_mappings(&mut self, src: RegionId, dsts: &[RegionId]) {
120        if let Some(dst_set) = self.src_to_dst.get_mut(&src) {
121            for dst in dsts {
122                dst_set.remove(dst);
123            }
124            if dst_set.is_empty() {
125                self.src_to_dst.remove(&src);
126            }
127        }
128    }
129}
130
131impl MetadataValue for TableRepartValue {
132    fn try_from_raw_value(raw_value: &[u8]) -> Result<Self> {
133        serde_json::from_slice::<TableRepartValue>(raw_value).context(SerdeJsonSnafu)
134    }
135
136    fn try_as_raw_value(&self) -> Result<Vec<u8>> {
137        serde_json::to_vec(self).context(SerdeJsonSnafu)
138    }
139}

141pub type TableRepartValueDecodeResult =
142    Result<Option<DeserializedValueWithBytes<TableRepartValue>>>;

144/// Decodes `KeyValue` to [TableRepartKey] and [TableRepartValue].
145pub fn table_repart_decoder(kv: KeyValue) -> Result<(TableRepartKey, TableRepartValue)> {
146    let key = TableRepartKey::from_bytes(&kv.key)?;
147    let value = TableRepartValue::try_from_raw_value(&kv.value)?;
148    Ok((key, value))
149}

160    /// Returns all table repartition entries.
161    pub async fn table_reparts(&self) -> Result<Vec<(TableId, TableRepartValue)>> {
162        let prefix = TableRepartKey::range_prefix();
163        let req = RangeRequest::new().with_prefix(prefix);
164        let stream = PaginationStream::new(
165            self.kv_backend.clone(),
166            req,
167            DEFAULT_PAGE_SIZE,
168            table_repart_decoder,
169        )
170        .into_stream();
171
172        let res = stream.try_collect::<Vec<_>>().await?;
173        Ok(res
174            .into_iter()
175            .map(|(key, value)| (key.table_id, value))
176            .collect())
177    }
178
179    /// Builds a create table repart transaction,
180    /// it expected the `__table_repart/{table_id}` wasn't occupied.
181    pub fn build_create_txn(
182        &self,
183        table_id: TableId,
184        table_repart_value: &TableRepartValue,
185    ) -> Result<(
186        Txn,
187        impl FnOnce(&mut TxnOpGetResponseSet) -> TableRepartValueDecodeResult + use<>,
188    )> {
189        let key = TableRepartKey::new(table_id);
190        let raw_key = key.to_bytes();
191
192        let txn = Txn::put_if_not_exists(raw_key.clone(), table_repart_value.try_as_raw_value()?);
193
194        Ok((
195            txn,
196            TxnOpGetResponseSet::decode_with(TxnOpGetResponseSet::filter(raw_key)),
197        ))
198    }
199
200    /// Builds a update table repart transaction,
201    /// it expected the remote value equals the `current_table_repart_value`.
202    /// It retrieves the latest value if the comparing failed.
203    pub fn build_update_txn(
204        &self,
205        table_id: TableId,
206        current_table_repart_value: &DeserializedValueWithBytes<TableRepartValue>,
207        new_table_repart_value: &TableRepartValue,
208    ) -> Result<(
209        Txn,
210        impl FnOnce(&mut TxnOpGetResponseSet) -> TableRepartValueDecodeResult + use<>,
211    )> {
212        let key = TableRepartKey::new(table_id);
213        let raw_key = key.to_bytes();
214        let raw_value = current_table_repart_value.get_raw_bytes();
215        let new_raw_value: Vec<u8> = new_table_repart_value.try_as_raw_value()?;
216
217        let txn = Txn::compare_and_put(raw_key.clone(), raw_value, new_raw_value);
218
219        Ok((
220            txn,
221            TxnOpGetResponseSet::decode_with(TxnOpGetResponseSet::filter(raw_key)),
222        ))
223    }

259    /// Returns batch of [`TableRepartValue`] that respects the order of `table_ids`.
260    pub async fn batch_get(&self, table_ids: &[TableId]) -> Result<Vec<Option<TableRepartValue>>> {
261        let raw_table_reparts = self.batch_get_inner(table_ids).await?;
262
263        Ok(raw_table_reparts
264            .into_iter()
265            .map(|v| v.map(|x| x.inner))
266            .collect())
267    }

270    pub async fn batch_get_with_raw_bytes(
271        &self,
272        table_ids: &[TableId],
273    ) -> Result<Vec<Option<DeserializedValueWithBytes<TableRepartValue>>>> {
274        self.batch_get_inner(table_ids).await
275    }
276
277    async fn batch_get_inner(
278        &self,
279        table_ids: &[TableId],
280    ) -> Result<Vec<Option<DeserializedValueWithBytes<TableRepartValue>>>> {
281        let keys = table_ids
282            .iter()
283            .map(|id| TableRepartKey::new(*id).to_bytes())
284            .collect::<Vec<_>>();
285        let resp = self
286            .kv_backend
287            .batch_get(BatchGetRequest { keys: keys.clone() })
288            .await?;
289
290        let kvs = resp
291            .kvs
292            .into_iter()
293            .map(|kv| (kv.key, kv.value))
294            .collect::<HashMap<_, _>>();
295        keys.into_iter()
296            .map(|key| {
297                if let Some(value) = kvs.get(&key) {
298                    Ok(Some(DeserializedValueWithBytes::from_inner_slice(value)?))
299                } else {
300                    Ok(None)
301                }
302            })
303            .collect()
304    }

306    /// Updates mappings from src region to dst regions.
307    /// Should be called once repartition is done.
308    pub async fn update_mappings(&self, src: RegionId, dst: &[RegionId]) -> Result<()> {
309        let table_id = src.table_id();
310
311        // Get current table repart with raw bytes for CAS operation
312        let current_table_repart = self
313            .get_with_raw_bytes(table_id)
314            .await?
315            .context(crate::error::TableRepartNotFoundSnafu { table_id })?;
316
317        // Clone the current repart value and update mappings
318        let mut new_table_repart_value = current_table_repart.inner.clone();
319        new_table_repart_value.update_mappings(src, dst);
320
321        // Execute atomic update
322        let (txn, _) =
323            self.build_update_txn(table_id, &current_table_repart, &new_table_repart_value)?;
324
325        let result = self.kv_backend.txn(txn).await?;
326
327        ensure!(
328            result.succeeded,
329            crate::error::MetadataCorruptionSnafu {
330                err_msg: format!(
331                    "Failed to update mappings for table {}: CAS operation failed",
332                    table_id
333                ),
334            }
335        );
336
337        Ok(())
338    }
339
340    /// Removes mappings from src region to dst regions.
341    /// Should be called once files from src region are cleaned up in dst regions.
342    pub async fn remove_mappings(&self, src: RegionId, dsts: &[RegionId]) -> Result<()> {
343        let table_id = src.table_id();
344
345        // Get current table repart with raw bytes for CAS operation
346        let current_table_repart = self
347            .get_with_raw_bytes(table_id)
348            .await?
349            .context(crate::error::TableRepartNotFoundSnafu { table_id })?;
350
351        // Clone the current repart value and remove mappings
352        let mut new_table_repart_value = current_table_repart.inner.clone();
353        new_table_repart_value.remove_mappings(src, dsts);
354
355        // Execute atomic update
356        let (txn, _) =
357            self.build_update_txn(table_id, &current_table_repart, &new_table_repart_value)?;
358
359        let result = self.kv_backend.txn(txn).await?;
360
361        ensure!(
362            result.succeeded,
363            crate::error::MetadataCorruptionSnafu {
364                err_msg: format!(
365                    "Failed to remove mappings for table {}: CAS operation failed",
366                    table_id
367                ),
368            }
369        );
370
371        Ok(())
372    }
373
374    /// Returns the destination regions for a given source region.
375    pub async fn get_dst_regions(
376        &self,
377        src_region: RegionId,
378    ) -> Result<Option<BTreeSet<RegionId>>> {
379        let table_id = src_region.table_id();
380        let table_repart = self.get(table_id).await?;
381        Ok(table_repart.and_then(|repart| repart.src_to_dst.get(&src_region).cloned()))
382    }
