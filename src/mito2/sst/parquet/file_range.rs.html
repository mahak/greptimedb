60/// Checks if a row group contains delete operations by examining the min value of op_type column.
61///
62/// Returns `Ok(true)` if the row group contains delete operations, `Ok(false)` if it doesn't,
63/// or an error if the statistics are not present or cannot be decoded.
64pub(crate) fn row_group_contains_delete(
65    parquet_meta: &ParquetMetaData,
66    row_group_index: usize,
67    file_path: &str,
68) -> Result<bool> {
69    let row_group_metadata = &parquet_meta.row_groups()[row_group_index];
70
71    // safety: The last column of SST must be op_type
72    let column_metadata = &row_group_metadata.columns().last().unwrap();
73    let stats = column_metadata
74        .statistics()
75        .context(StatsNotPresentSnafu { file_path })?;
76    stats
77        .min_bytes_opt()
78        .context(StatsNotPresentSnafu { file_path })?
79        .try_into()
80        .map(i32::from_le_bytes)
81        .map(|min_op_type| min_op_type == OpType::Delete as i32)
82        .ok()
83        .context(DecodeStatsSnafu { file_path })
84}

86/// A range of a parquet SST. Now it is a row group.
87/// We can read different file ranges in parallel.
88#[derive(Clone)]
89pub struct FileRange {
90    /// Shared context.
91    context: FileRangeContextRef,
92    /// Index of the row group in the SST.
93    row_group_idx: usize,
94    /// Row selection for the row group. `None` means all rows.
95    row_selection: Option<RowSelection>,
96}
97
98impl FileRange {
99    /// Creates a new [FileRange].
100    pub(crate) fn new(
101        context: FileRangeContextRef,
102        row_group_idx: usize,
103        row_selection: Option<RowSelection>,
104    ) -> Self {
105        Self {
106            context,
107            row_group_idx,
108            row_selection,
109        }
110    }
111
112    /// Returns true if [FileRange] selects all rows in row group.
113    fn select_all(&self) -> bool {
114        let rows_in_group = self
115            .context
116            .reader_builder
117            .parquet_metadata()
118            .row_group(self.row_group_idx)
119            .num_rows();
120
121        let Some(row_selection) = &self.row_selection else {
122            return true;
123        };
124        row_selection.row_count() == rows_in_group as usize
125    }
126
127    /// Performs pruning before reading the [FileRange].
128    /// It use latest dynamic filters with row group statistics to prune the range.
129    ///
130    /// Returns false if the entire range is pruned and can be skipped.
131    fn in_dynamic_filter_range(&self) -> bool {
132        if self.context.base.dyn_filters.is_empty() {
133            return true;
134        }
135        let curr_row_group = self
136            .context
137            .reader_builder
138            .parquet_metadata()
139            .row_group(self.row_group_idx);
140        let read_format = self.context.read_format();
141        let prune_schema = &self.context.base.prune_schema;
142        let stats = RowGroupPruningStats::new(
143            std::slice::from_ref(curr_row_group),
144            read_format,
145            self.context.base.expected_metadata.clone(),
146            self.compute_skip_fields(),
147        );
148
149        // not costly to create a predicate here since dynamic filters are wrapped in Arc
150        let pred = Predicate::new(vec![]).with_dyn_filters(self.context.base.dyn_filters.clone());
151
152        pred.prune_with_stats(&stats, prune_schema.arrow_schema())
153            .first()
154            .cloned()
155            .unwrap_or(true) // unexpected, not skip just in case
156    }
157
158    fn compute_skip_fields(&self) -> bool {
159        match self.context.base.pre_filter_mode {
160            PreFilterMode::All => false,
161            PreFilterMode::SkipFields => true,
162            PreFilterMode::SkipFieldsOnDelete => {
163                // Check if this specific row group contains delete op
164                row_group_contains_delete(
165                    self.context.reader_builder.parquet_metadata(),
166                    self.row_group_idx,
167                    self.context.reader_builder.file_path(),
168                )
169                .unwrap_or(true)
170            }
171        }
172    }
173
174    /// Returns a reader to read the [FileRange].
175    pub(crate) async fn reader(
176        &self,
177        selector: Option<TimeSeriesRowSelector>,
178        fetch_metrics: Option<&ParquetFetchMetrics>,
179    ) -> Result<Option<PruneReader>> {
180        if !self.in_dynamic_filter_range() {
181            return Ok(None);
182        }
183        let parquet_reader = self
184            .context
185            .reader_builder
186            .build(
187                self.row_group_idx,
188                self.row_selection.clone(),
189                fetch_metrics,
190            )
191            .await?;
192
193        let use_last_row_reader = if selector
194            .map(|s| s == TimeSeriesRowSelector::LastRow)
195            .unwrap_or(false)
196        {
197            // Only use LastRowReader if row group does not contain DELETE
198            // and all rows are selected.
199            let put_only = !self
200                .context
201                .contains_delete(self.row_group_idx)
202                .inspect_err(|e| {
203                    error!(e; "Failed to decode min value of op_type, fallback to RowGroupReader");
204                })
205                .unwrap_or(true);
206            put_only && self.select_all()
207        } else {
208            // No selector provided, use RowGroupReader
209            false
210        };
211
212        // Compute skip_fields once for this row group
213        let skip_fields = self.context.should_skip_fields(self.row_group_idx);
214
215        let prune_reader = if use_last_row_reader {
216            // Row group is PUT only, use LastRowReader to skip unnecessary rows.
217            let reader = RowGroupLastRowCachedReader::new(
218                self.file_handle().file_id().file_id(),
219                self.row_group_idx,
220                self.context.reader_builder.cache_strategy().clone(),
221                RowGroupReader::new(self.context.clone(), parquet_reader),
222            );
223            PruneReader::new_with_last_row_reader(self.context.clone(), reader, skip_fields)
224        } else {
225            // Row group contains DELETE, fallback to default reader.
226            PruneReader::new_with_row_group_reader(
227                self.context.clone(),
228                RowGroupReader::new(self.context.clone(), parquet_reader),
229                skip_fields,
230            )
231        };
232
233        Ok(Some(prune_reader))
234    }
235
236    /// Creates a flat reader that returns RecordBatch.
237    pub(crate) async fn flat_reader(
238        &self,
239        fetch_metrics: Option<&ParquetFetchMetrics>,
240    ) -> Result<Option<FlatPruneReader>> {
241        if !self.in_dynamic_filter_range() {
242            return Ok(None);
243        }
244        let parquet_reader = self
245            .context
246            .reader_builder
247            .build(
248                self.row_group_idx,
249                self.row_selection.clone(),
250                fetch_metrics,
251            )
252            .await?;
253
254        // Compute skip_fields once for this row group
255        let skip_fields = self.context.should_skip_fields(self.row_group_idx);
256
257        let flat_row_group_reader = FlatRowGroupReader::new(self.context.clone(), parquet_reader);
258        let flat_prune_reader = FlatPruneReader::new_with_row_group_reader(
259            self.context.clone(),
260            flat_row_group_reader,
261            skip_fields,
262        );
263
264        Ok(Some(flat_prune_reader))
265    }

270    }

273    pub(crate) fn file_handle(&self) -> &FileHandle {
274        self.context.reader_builder.file_handle()
275    }

429impl RangeBase {
430    /// TRY THE BEST to perform pushed down predicate precisely on the input batch.
431    /// Return the filtered batch. If the entire batch is filtered out, return None.
432    ///
433    /// Supported filter expr type is defined in [SimpleFilterEvaluator].
434    ///
435    /// When a filter is referencing primary key column, this method will decode
436    /// the primary key and put it into the batch.
437    ///
438    /// # Arguments
439    /// * `input` - The batch to filter
440    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
441    pub(crate) fn precise_filter(
442        &self,
443        mut input: Batch,
444        skip_fields: bool,
445    ) -> Result<Option<Batch>> {
446        let mut mask = BooleanBuffer::new_set(input.num_rows());
447
448        // Run filter one by one and combine them result
449        // TODO(ruihang): run primary key filter first. It may short circuit other filters
450        for filter_ctx in &self.filters {
451            let filter = match filter_ctx.filter() {
452                MaybeFilter::Filter(f) => f,
453                // Column matches.
454                MaybeFilter::Matched => continue,
455                // Column doesn't match, filter the entire batch.
456                MaybeFilter::Pruned => return Ok(None),
457            };
458            let result = match filter_ctx.semantic_type() {
459                SemanticType::Tag => {
460                    let pk_values = if let Some(pk_values) = input.pk_values() {
461                        pk_values
462                    } else {
463                        input.set_pk_values(
464                            self.codec
465                                .decode(input.primary_key())
466                                .context(DecodeSnafu)?,
467                        );
468                        input.pk_values().unwrap()
469                    };
470                    let pk_value = match pk_values {
471                        CompositeValues::Dense(v) => {
472                            // Safety: this is a primary key
473                            let pk_index = self
474                                .read_format
475                                .metadata()
476                                .primary_key_index(filter_ctx.column_id())
477                                .unwrap();
478                            v[pk_index]
479                                .1
480                                .try_to_scalar_value(filter_ctx.data_type())
481                                .context(DataTypeMismatchSnafu)?
482                        }
483                        CompositeValues::Sparse(v) => {
484                            let v = v.get_or_null(filter_ctx.column_id());
485                            v.try_to_scalar_value(filter_ctx.data_type())
486                                .context(DataTypeMismatchSnafu)?
487                        }
488                    };
489                    if filter
490                        .evaluate_scalar(&pk_value)
491                        .context(RecordBatchSnafu)?
492                    {
493                        continue;
494                    } else {
495                        // PK not match means the entire batch is filtered out.
496                        return Ok(None);
497                    }
498                }
499                SemanticType::Field => {
500                    // Skip field filters if skip_fields is true
501                    if skip_fields {
502                        continue;
503                    }
504                    // Safety: Input is Batch so we are using primary key format.
505                    let Some(field_index) = self
506                        .read_format
507                        .as_primary_key()
508                        .unwrap()
509                        .field_index_by_id(filter_ctx.column_id())
510                    else {
511                        continue;
512                    };
513                    let field_col = &input.fields()[field_index].data;
514                    filter
515                        .evaluate_vector(field_col)
516                        .context(RecordBatchSnafu)?
517                }
518                SemanticType::Timestamp => filter
519                    .evaluate_vector(input.timestamps())
520                    .context(RecordBatchSnafu)?,
521            };
522
523            mask = mask.bitand(&result);
524        }
525
526        if mask.count_set_bits() == 0 {
527            return Ok(None);
528        }
529
530        // Apply partition filter
531        if let Some(partition_filter) = &self.partition_filter {
532            let record_batch = self
533                .build_record_batch_for_pruning(&mut input, &partition_filter.partition_schema)?;
534            let partition_mask = self.evaluate_partition_filter(&record_batch, partition_filter)?;
535            mask = mask.bitand(&partition_mask);
536        }
537
538        if mask.count_set_bits() == 0 {
539            Ok(None)
540        } else {
541            input.filter(&BooleanArray::from(mask).into())?;
542            Ok(Some(input))
543        }
544    }
545
546    /// Filters the input RecordBatch by the pushed down predicate and returns RecordBatch.
547    ///
548    /// It assumes all necessary tags are already decoded from the primary key.
549    ///
550    /// # Arguments
551    /// * `input` - The RecordBatch to filter
552    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
553    pub(crate) fn precise_filter_flat(
554        &self,
555        input: RecordBatch,
556        skip_fields: bool,
557    ) -> Result<Option<RecordBatch>> {
558        let mut tag_decode_state = TagDecodeState::new();
559        let mask = self.compute_filter_mask_flat(&input, skip_fields, &mut tag_decode_state)?;
560
561        // If mask is None, the entire batch is filtered out
562        let Some(mut mask) = mask else {
563            return Ok(None);
564        };
565
566        // Apply partition filter
567        if let Some(partition_filter) = &self.partition_filter {
568            let record_batch = self.project_record_batch_for_pruning_flat(
569                &input,
570                &partition_filter.partition_schema,
571                &mut tag_decode_state,
572            )?;
573            let partition_mask = self.evaluate_partition_filter(&record_batch, partition_filter)?;
574            mask = mask.bitand(&partition_mask);
575        }
576
577        if mask.count_set_bits() == 0 {
578            return Ok(None);
579        }
580
581        let filtered_batch =
582            datatypes::arrow::compute::filter_record_batch(&input, &BooleanArray::from(mask))
583                .context(ComputeArrowSnafu)?;
584
585        if filtered_batch.num_rows() > 0 {
586            Ok(Some(filtered_batch))
587        } else {
588            Ok(None)
589        }
590    }
591
592    /// Computes the filter mask for the input RecordBatch based on pushed down predicates.
593    /// If a partition expr filter is configured, it is applied later in `precise_filter_flat` but **NOT** in this function.
594    ///
595    /// Returns `None` if the entire batch is filtered out, otherwise returns the boolean mask.
596    ///
597    /// # Arguments
598    /// * `input` - The RecordBatch to compute mask for
599    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
600    pub(crate) fn compute_filter_mask_flat(
601        &self,
602        input: &RecordBatch,
603        skip_fields: bool,
604        tag_decode_state: &mut TagDecodeState,
605    ) -> Result<Option<BooleanBuffer>> {
606        let mut mask = BooleanBuffer::new_set(input.num_rows());
607
608        let flat_format = self
609            .read_format
610            .as_flat()
611            .context(crate::error::UnexpectedSnafu {
612                reason: "Expected flat format for precise_filter_flat",
613            })?;
614        let metadata = flat_format.metadata();
615
616        // Run filter one by one and combine them result
617        for filter_ctx in &self.filters {
618            let filter = match filter_ctx.filter() {
619                MaybeFilter::Filter(f) => f,
620                // Column matches.
621                MaybeFilter::Matched => continue,
622                // Column doesn't match, filter the entire batch.
623                MaybeFilter::Pruned => return Ok(None),
624            };
625
626            // Skip field filters if skip_fields is true
627            if skip_fields && filter_ctx.semantic_type() == SemanticType::Field {
628                continue;
629            }
630
631            // Get the column directly by its projected index
632            let column_idx = flat_format.projected_index_by_id(filter_ctx.column_id());
633            if let Some(idx) = column_idx {
634                let column = &input.columns()[idx];
635                let result = filter.evaluate_array(column).context(RecordBatchSnafu)?;
636                mask = mask.bitand(&result);
637            } else if filter_ctx.semantic_type() == SemanticType::Tag {
638                // Column not found in projection, it may be a tag column.
639                let column_id = filter_ctx.column_id();
640
641                if let Some(tag_column) =
642                    self.maybe_decode_tag_column(metadata, column_id, input, tag_decode_state)?
643                {
644                    let result = filter
645                        .evaluate_array(&tag_column)
646                        .context(RecordBatchSnafu)?;
647                    mask = mask.bitand(&result);
648                }
649            } else if filter_ctx.semantic_type() == SemanticType::Timestamp {
650                let time_index_pos = time_index_column_index(input.num_columns());
651                let column = &input.columns()[time_index_pos];
652                let result = filter.evaluate_array(column).context(RecordBatchSnafu)?;
653                mask = mask.bitand(&result);
654            }
655            // Non-tag column not found in projection.
656        }
657
658        Ok(Some(mask))
659    }
660
661    /// Returns the decoded tag column for `column_id`, or `None` if it's not a tag.
662    fn maybe_decode_tag_column(
663        &self,
664        metadata: &RegionMetadataRef,
665        column_id: ColumnId,
666        input: &RecordBatch,
667        tag_decode_state: &mut TagDecodeState,
668    ) -> Result<Option<ArrayRef>> {
669        let Some(pk_index) = metadata.primary_key_index(column_id) else {
670            return Ok(None);
671        };
672
673        if let Some(cached_column) = tag_decode_state.decoded_tag_cache.get(&column_id) {
674            return Ok(Some(cached_column.clone()));
675        }
676
677        if tag_decode_state.decoded_pks.is_none() {
678            tag_decode_state.decoded_pks = Some(decode_primary_keys(self.codec.as_ref(), input)?);
679        }
680
681        let pk_index = if self.codec.encoding() == PrimaryKeyEncoding::Sparse {
682            None
683        } else {
684            Some(pk_index)
685        };
686        let Some(column_index) = metadata.column_index_by_id(column_id) else {
687            return Ok(None);
688        };
689        let Some(decoded) = tag_decode_state.decoded_pks.as_ref() else {
690            return Ok(None);
691        };
692
693        let column_metadata = &metadata.column_metadatas[column_index];
694        let tag_column = decoded.get_tag_column(
695            column_id,
696            pk_index,
697            &column_metadata.column_schema.data_type,
698        )?;
699        tag_decode_state
700            .decoded_tag_cache
701            .insert(column_id, tag_column.clone());
702
703        Ok(Some(tag_column))
704    }

706    /// Evaluates the partition filter against the input `RecordBatch`.
707    fn evaluate_partition_filter(
708        &self,
709        record_batch: &RecordBatch,
710        partition_filter: &PartitionFilterContext,
711    ) -> Result<BooleanBuffer> {
712        let columnar_value = partition_filter
713            .region_partition_physical_expr
714            .evaluate(record_batch)
715            .context(EvalPartitionFilterSnafu)?;
716        let array = columnar_value
717            .into_array(record_batch.num_rows())
718            .context(EvalPartitionFilterSnafu)?;
719        let boolean_array =
720            array
721                .as_any()
722                .downcast_ref::<BooleanArray>()
723                .context(UnexpectedSnafu {
724                    reason: "Failed to downcast to BooleanArray".to_string(),
725                })?;
726
727        Ok(boolean_array.values().clone())
728    }

730    /// Builds a `RecordBatch` from the input `Batch` matching the given schema.
731    ///
732    /// This is used for partition expression evaluation. The schema should only contain
733    /// the columns referenced by the partition expression to minimize overhead.
734    fn build_record_batch_for_pruning(
735        &self,
736        input: &mut Batch,
737        schema: &Arc<Schema>,
738    ) -> Result<RecordBatch> {
739        let arrow_schema = schema.arrow_schema();
740        let mut columns = Vec::with_capacity(arrow_schema.fields().len());
741
742        // Decode primary key if necessary.
743        if input.pk_values().is_none() {
744            input.set_pk_values(
745                self.codec
746                    .decode(input.primary_key())
747                    .context(DecodeSnafu)?,
748            );
749        }
750
751        for field in arrow_schema.fields() {
752            let metadata = self.read_format.metadata();
753            let column_id = metadata.column_by_name(field.name()).map(|c| c.column_id);
754
755            // Partition pruning schema should be a subset of the input batch schema.
756            let Some(column_id) = column_id else {
757                return UnexpectedSnafu {
758                    reason: format!(
759                        "Partition pruning schema expects column '{}' but it is missing in \
760                         region metadata",
761                        field.name()
762                    ),
763                }
764                .fail();
765            };
766
767            // 1. Check if it's a tag.
768            if let Some(pk_index) = metadata.primary_key_index(column_id) {
769                let pk_values = input.pk_values().unwrap();
770                let value = match pk_values {
771                    CompositeValues::Dense(v) => &v[pk_index].1,
772                    CompositeValues::Sparse(v) => v.get_or_null(column_id),
773                };
774                let concrete_type = ConcreteDataType::from_arrow_type(field.data_type());
775                let arrow_scalar = value
776                    .try_to_scalar_value(&concrete_type)
777                    .context(DataTypeMismatchSnafu)?;
778                let array = arrow_scalar
779                    .to_array_of_size(input.num_rows())
780                    .context(EvalPartitionFilterSnafu)?;
781                columns.push(array);
782            } else if metadata.time_index_column().column_id == column_id {
783                // 2. Check if it's the timestamp column.
784                columns.push(input.timestamps().to_arrow_array());
785            } else if let Some(field_index) = self
786                .read_format
787                .as_primary_key()
788                .and_then(|f| f.field_index_by_id(column_id))
789            {
790                // 3. Check if it's a field column.
791                columns.push(input.fields()[field_index].data.to_arrow_array());
792            } else {
793                return UnexpectedSnafu {
794                    reason: format!(
795                        "Partition pruning schema expects column '{}' (id {}) but it is not \
796                         present in input batch",
797                        field.name(),
798                        column_id
799                    ),
800                }
801                .fail();
802            }
803        }
804
805        RecordBatch::try_new(arrow_schema.clone(), columns).context(NewRecordBatchSnafu)
806    }

808    /// Projects the input `RecordBatch` to match the given schema.
809    ///
810    /// This is used for partition expression evaluation. The schema should only contain
811    /// the columns referenced by the partition expression to minimize overhead.
812    fn project_record_batch_for_pruning_flat(
813        &self,
814        input: &RecordBatch,
815        schema: &Arc<Schema>,
816        tag_decode_state: &mut TagDecodeState,
817    ) -> Result<RecordBatch> {
818        let arrow_schema = schema.arrow_schema();
819        let mut columns = Vec::with_capacity(arrow_schema.fields().len());
820
821        let flat_format = self
822            .read_format
823            .as_flat()
824            .context(crate::error::UnexpectedSnafu {
825                reason: "Expected flat format for precise_filter_flat",
826            })?;
827        let metadata = flat_format.metadata();
828
829        for field in arrow_schema.fields() {
830            let column_id = metadata.column_by_name(field.name()).map(|c| c.column_id);
831
832            let Some(column_id) = column_id else {
833                return UnexpectedSnafu {
834                    reason: format!(
835                        "Partition pruning schema expects column '{}' but it is missing in \
836                         region metadata",
837                        field.name()
838                    ),
839                }
840                .fail();
841            };
842
843            if let Some(idx) = flat_format.projected_index_by_id(column_id) {
844                columns.push(input.column(idx).clone());
845                continue;
846            }
847
848            if metadata.time_index_column().column_id == column_id {
849                let time_index_pos = time_index_column_index(input.num_columns());
850                columns.push(input.column(time_index_pos).clone());
851                continue;
852            }
853
854            if let Some(tag_column) =
855                self.maybe_decode_tag_column(metadata, column_id, input, tag_decode_state)?
856            {
857                columns.push(tag_column);
858                continue;
859            }
860
861            return UnexpectedSnafu {
862                reason: format!(
863                    "Partition pruning schema expects column '{}' (id {}) but it is not \
864                     present in projected record batch",
865                    field.name(),
866                    column_id
867                ),
868            }
869            .fail();
870        }
871
872        RecordBatch::try_new(arrow_schema.clone(), columns).context(NewRecordBatchSnafu)
873    }
